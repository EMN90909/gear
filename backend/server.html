const express = require("express"); const fileUpload = require("express-fileupload"); const { exec } = require("child_process"); const app = express(); app.use(fileUpload()); app.post("/transcribe", (req, res) => { if (!req.files || !req.files.audio) return res.status(400).send("No file uploaded."); const audioFile = req.files.audio; const path = `./uploads/${audioFile.name}`; audioFile.mv(path, (err) => { if (err) return res.status(500).send(err); // Call whisper.cpp exec(`./whisper.cpp/main -f ${path}`, (err, stdout, stderr) => { if (err) return res.status(500).send(stderr); res.json({ text: stdout }); }); }); }); app.listen(3000, () => console.log("Server running on port 3000"));
